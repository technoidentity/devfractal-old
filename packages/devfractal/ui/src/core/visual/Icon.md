### Default Icon

```jsx
import { faUser } from '@fortawesome/free-solid-svg-icons'
;<Icon icon={faUser} />
```

### Icon with all properties

```jsx
import { faUser } from '@fortawesome/free-solid-svg-icons'
;<Icon
  icon={faUser}
  size="1x"
  fixedWidth={false}
  border={false}
  spin={false}
  pulse={false}
  color="green"
  transform="flip-h"
/>
```
